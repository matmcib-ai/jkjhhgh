-- Draggable
frame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        local startPos = frame.Position
        local dragStart = input.Position
       
        local conn = game:GetService("UserInputService").InputChanged:Connect(function(inp)
            if inp.UserInputType == Enum.UserInputType.MouseMovement then
                local delta = inp.Position - dragStart
                frame.Position = UDim2.new(
                    startPos.X.Scale, startPos.X.Offset + delta.X,
                    startPos.Y.Scale, startPos.Y.Offset + delta.Y
                )
            end
        end)
       
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                conn:Disconnect()
            end
        end)
    end
end)

-- ==================== DESYNC HOOK (Zyro Style) ====================
local lastUpdate = os.clock()
local updateInterval = 4.0

local function rakhook(packet)
    if packet.PacketId == 0x1B then
        local data = packet.AsBuffer
        
        if os.clock() - lastUpdate >= updateInterval then
            lastUpdate = os.clock()
            -- Update real position every 4 seconds
        else
            buffer.writeu32(data, 1, 0/0) -- NaN Desync
            packet:SetData(data)
        end
    end
end

pcall(function() raknet.add_send_hook(rakhook) end)
